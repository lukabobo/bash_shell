# Lệnh `true`, `false` và `:`

## 11.1. Vòng lặp vô hạn

    while true; do
        echo ok
    done

or

    while :; do
        echo ok
    done

or

    until false; do
        echo ok
    done

## 11.2. Chức năng trả về 

    function positive() {
        return 0
    }

    function negative() {
        return 1
    }

## 11.3. Code sẽ luôn luôn thực thi hoặc không bao giờ thực thi

    if true; then
        echo Always executed
    fi
    if false; then
        echo Never executed
    fi
